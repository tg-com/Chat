const socket = io(); // --- DOM Elements --- const loginScreen = document.getElementById('login-screen'); const chatScreen = document.getElementById('chat-screen'); const loginForm = document.getElementById('login-form'); const usernameInput = document.getElementById('username'); const roomInput = document.getElementById('room-ip'); const statusDiv = document.getElementById('status'); const btnText = document.getElementById('btn-text'); const btnLoader = document.getElementById('btn-loader'); const messagesDiv = document.getElementById('messages'); const messageInput = document.getElementById('message-input'); const sendBtn = document.getElementById('send-btn'); const displayRoom = document.getElementById('display-room'); const displayUsername = document.getElementById('display-username'); const leaveBtn = document.getElementById('leave-btn'); // Naya Element: Typing aur Online Users Status ke liye const typingDiv = document.createElement('div'); typingDiv.id = "typing-indicator"; typingDiv.style.fontSize = "12px"; typingDiv.style.color = "gray"; messagesDiv.after(typingDiv); const onlineCountDiv = document.createElement('div'); onlineCountDiv.style.fontSize = "12px"; onlineCountDiv.style.color = "#25D366"; displayRoom.after(onlineCountDiv); let currentUser = ''; let currentRoom = ''; // --- 1. Login Handle (No Password) --- loginForm.addEventListener('submit', (e) => { e.preventDefault(); const username = usernameInput.value.trim(); const room = roomInput.value.trim(); if (!username || !room) return; currentUser = username; currentRoom = room; btnText.classList.add('hidden'); btnLoader.classList.remove('hidden'); socket.emit('join-room', { username, room }); }); // --- 2. Live Status & Joined confirmation --- socket.on('joined-room', ({ room, username }) => { btnText.classList.remove('hidden'); btnLoader.classList.add('hidden'); loginScreen.classList.remove('active'); chatScreen.classList.add('active'); displayRoom.textContent = room; displayUsername.textContent = username; messagesDiv.innerHTML = ''; addSystemMessage(`Welcome to room "${room}"`); }); // Room mein kitne log hain update karna socket.on('update-user-count', (count) => { onlineCountDiv.textContent = `Online: ${count}`; }); // --- 3. Typing Indicator Logic --- messageInput.addEventListener('input', () => { socket.emit('typing', { room: currentRoom, username: currentUser }); }); socket.on('display-typing', (data) => { if (data.username !== currentUser) { typingDiv.textContent = data.isTyping ? `${data.username} is typing...` : ''; } }); // --- 4. New Message (Blue Tick & 60s Auto-Delete) --- socket.on('new-message', ({ username, message, timestamp }) => { const isOwn = username === currentUser; const msgElement = addMessage(username, message, isOwn, timestamp); // Seen status bhejona agar hamne message padha if (!isOwn) { socket.emit('mark-seen', { room: currentRoom, msgId: timestamp }); } // FEATURE: 60 Second Auto-Delete setTimeout(() => { if (msgElement) { msgElement.style.opacity = '0'; msgElement.style.transform = 'translateX(20px)'; setTimeout(() => msgElement.remove(), 500); } }, 60000); }); // Blue Tick Update karna socket.on('message-seen-update', (msgId) => { const ticks = document.querySelector(`[data-id="${msgId}"] .ticks`); if (ticks) { ticks.textContent = ' ✔✔'; ticks.style.color = '#34B7F1'; } }); // --- 5. Cleanup & Safety --- function cleanupAndLeave() { if (currentRoom) { socket.emit('leave-room', { room: currentRoom }); } messagesDiv.innerHTML = ''; chatScreen.classList.remove('active'); loginScreen.classList.add('active'); window.location.reload(); // Hard reset for memory safety } leaveBtn.addEventListener('click', cleanupAndLeave); window.addEventListener('beforeunload', cleanupAndLeave); // --- Helper Functions --- function addMessage(username, message, isOwn, timestamp) { const div = document.createElement('div'); div.className = `message ${isOwn ? 'own' : 'other'}`; div.setAttribute('data-id', timestamp); div.style.transition = 'all 0.5s ease-out'; const time = new Date(timestamp).toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'}); const ticks = isOwn ? ' ✔' : ''; div.innerHTML = `
${username} • ${time} ${ticks}
${escapeHtml(message)}
`; messagesDiv.appendChild(div); scrollToBottom(); return div; } function addSystemMessage(text) { const div = document.createElement('div'); div.className = 'message system'; div.textContent = text; messagesDiv.appendChild(div); setTimeout(() => div.remove(), 5000); scrollToBottom(); } function scrollToBottom() { messagesDiv.scrollTop = messagesDiv.scrollHeight; } function escapeHtml(text) { const div = document.createElement('div'); div.textContent = text; return div.innerHTML; }
